SiebelAppFacade.RatingControl = (function() {
  var MY_ID = "RatingControl";
  var properties = {
    rating: 0,
    maxRating: 5,
    readOnly: false
  };

  return {
    Init: function(options) {
      properties = options || properties;
    },

    Render: function(tagId) {
      var stars = '';
      for (var i = 1; i <= properties.maxRating; i++) {
        var starClass = i <= properties.rating ? 'filled' : '';
        stars += '<span class="star ' + starClass + '">â˜…</span>';
      }
      document.getElementById(tagId).innerHTML = 
        '<div class="rating">' + stars + '</div>';
      
      // Add event listeners
      if (!properties.readOnly) {
        var stars = document.querySelectorAll('.star');
        stars.forEach(function(star, index) {
          star.addEventListener('click', function() {
            properties.rating = index + 1;
            RatingControl.Render(tagId);
          });
        });
      }
    },

    SetProperty: function(name, value) {
      if (properties.hasOwnProperty(name)) {
        properties[name] = value;
      }
    },

    GetProperty: function(name) {
      return properties[name];
    }
  };
})();

// Custom Control with Events
SiebelAppFacade.MyControl = (function() {
  return {
    Init: function(options) {
      // Setup event listeners
      jQuery('#saveBtn').on('click', function() {
        MyControl.Save();
      });
      
      jQuery('#deleteBtn').on('click', function() {
        if (confirm('Are you sure?')) {
          MyControl.Delete();
        }
      });
    },

    Save: function() {
      var data = {
        name: jQuery('#nameInput').val(),
        description: jQuery('#descInput').val()
      };
      
      // Validate
      if (!data.name) {
        alert('Name is required');
        return false;
      }
      
      // Call Siebel method
      var bc = GetBusComp('Account');
      bc.SetFieldValue('Name', data.name);
      bc.SetFieldValue('Description', data.description);
      bc.WriteRecord();
      
      alert('Saved successfully!');
    },

    Delete: function() {
      var bc = GetBusComp('Account');
      bc.DeleteRecord();
      alert('Record deleted!');
    }
  };
})();
